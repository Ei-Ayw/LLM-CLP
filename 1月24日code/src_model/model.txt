import torch
from torch.nn import functional as F
from transformers import BertTokenizer, BertModel
from sklearn.model_selection import train_test_split
from torch.utils.data import DataLoader, Dataset
import pandas as pd
import torch.nn as nn
from sklearn.metrics import accuracy_score, recall_score, f1_score
from tqdm import tqdm

# 加载数据
def load_data(file1, file2):
    df1 = pd.read_excel(file1, usecols=[0])
    df1['label'] = 0
    df2 = pd.read_excel(file2, usecols=[0])
    df2['label'] = 1
    df = pd.concat([df1, df2])
    return df

# 分割数据集
def split_dataset(df):
    X_train, X_val, y_train, y_val = train_test_split(df.text.values, df.label.values, test_size=0.1, random_state=2020)
    return X_train, X_val, y_train, y_val

#  数据预处理
def preprocess_data(tokenizer, texts):
    encodings = {'input_ids': [], 'attention_mask': [], 'token_type_ids': []}
    for text in texts:
        encoding = tokenizer.encode_plus(text, padding='max_length', truncation=True, max_length=128, return_tensors='pt')
        encodings['input_ids'].append(encoding['input_ids'])
        encodings['attention_mask'].append(encoding['attention_mask'])
        encodings['token_type_ids'].append(encoding['token_type_ids'])
    return encodings


class CustomDataset(Dataset):
    def __init__(self, encodings, labels):
        self.encodings = {key: torch.cat(val, dim=0) for key, val in encodings.items()}
        self.labels = labels

    def __getitem__(self, idx):
        item = {key: val[idx] for key, val in self.encodings.items()}
        item['labels'] = torch.tensor(self.labels[idx])
        return item

    def __len__(self):
        return len(self.labels)


# 创建模型
class CNN_BiLSTM(nn.Module):
    def __init__(self):
        super(CNN_BiLSTM, self).__init__()
        self.cnn = nn.Conv1d(in_channels=768, out_channels=512, kernel_size=3, stride=1, padding=1)
        self.relu = nn.ReLU()
        self.lstm = nn.LSTM(input_size=512, hidden_size=256, num_layers=2, batch_first=True, bidirectional=True)
        self.fc = nn.Linear(256*2, 2)

    def forward(self, x):
        x = x.permute(0, 2, 1)
        x = self.cnn(x)
        x = self.relu(x)
        x = x.permute(0, 2, 1)
        x,_ = self.lstm(x)
        x = self.fc(x[:, -1, :])
        return x

class BertClassifier(nn.Module):
    def __init__(self):
        super(BertClassifier, self).__init__()
        self.bert_model = BertModel.from_pretrained('bert-base-chinese')
        self.cnn_bilstm = CNN_BiLSTM()

    def forward(self, input_ids, attention_mask):
        output = self.bert_model(input_ids=input_ids, attention_mask=attention_mask)
        sequence_output = output[0]
        output = self.cnn_bilstm(sequence_output)
        output = F.log_softmax(output, dim=1)
        return output


# 主函数
def main():
    print("Starting the program...")
    file1 = r"C:\Users\ssy\Desktop\111\支撑材料\第一问\train_insensitiveness.xlsx"
    file2 = r"C:\Users\ssy\Desktop\111\支撑材料\第一问\train_sensitiveness.xlsx"

    print("Loading data...")
    df = load_data(file1, file2)

    print("Splitting dataset...")
    X_train, X_val, y_train, y_val = split_dataset(df)

    print("Initializing tokenizer...")
    tokenizer = BertTokenizer.from_pretrained('bert-base-chinese')

    print("Preprocessing data...")
    train_encodings = preprocess_data(tokenizer, X_train)
    val_encodings = preprocess_data(tokenizer, X_val)

    print("Creating datasets...")
    train_dataset = CustomDataset(train_encodings, y_train)
    val_dataset = CustomDataset(val_encodings, y_val)

    print("Creating model...")
    model = BertClassifier()
    model = model.to('cuda')

    print("Creating data loaders...")
    train_loader = DataLoader(train_dataset, batch_size=64, shuffle=True)
    val_loader = DataLoader(val_dataset, batch_size=64, shuffle=True)

    print("Setting up training...")
    criterion = nn.NLLLoss()
    optimizer = torch.optim.Adam(params=model.parameters(), lr=1e-5)

    best_val_loss = float('inf')
    no_improve_count = 0  # Counter for epochs without improvement
    for epoch in range(10):  # 训练更多的epoch
        model.train()
        train_loss = 0
        train_outputs = []
        train_targets = []
        for _, data in enumerate(train_loader, 0):
            ids = data['input_ids'].to('cuda', dtype=torch.long)
            mask = data['attention_mask'].to('cuda', dtype=torch.long)
            targets = data['labels'].to('cuda', dtype=torch.long)

            outputs = model(ids, mask)
            loss = criterion(outputs, targets)
            train_loss += loss.item()
            optimizer.zero_grad()
            loss.backward()
            optimizer.step()

            train_outputs.extend(torch.argmax(outputs, dim=1).cpu().numpy())
            train_targets.extend(targets.cpu().numpy())

        train_loss /= len(train_loader)
        train_accuracy = accuracy_score(train_targets, train_outputs)
        train_recall = recall_score(train_targets, train_outputs, average='macro')
        train_f1 = f1_score(train_targets, train_outputs, average='macro')

        # 在每个epoch后评估模型在验证集上的性能
        model.eval()
        val_loss = 0
        val_outputs = []
        val_targets = []
        with torch.no_grad():
            for _, data in enumerate(val_loader, 0):
                ids = data['input_ids'].to('cuda', dtype=torch.long)
                mask = data['attention_mask'].to('cuda', dtype=torch.long)
                targets = data['labels'].to('cuda', dtype=torch.long)

                outputs = model(ids, mask)
                val_loss += criterion(outputs, targets).item()

                val_outputs.extend(torch.argmax(outputs, dim=1).cpu().numpy())
                val_targets.extend(targets.cpu().numpy())

        val_loss /= len(val_loader)
        val_accuracy = accuracy_score(val_targets, val_outputs)
        val_recall = recall_score(val_targets, val_outputs, average='macro')
        val_f1 = f1_score(val_targets, val_outputs, average='macro')

        print(f'Epoch {epoch + 1}, Train Loss: {train_loss:.4f}, Val Loss: {val_loss:.4f}')
        print(f'Train Accuracy: {train_accuracy:.4f}, Val Accuracy: {val_accuracy:.4f}')
        print(f'Train Recall: {train_recall:.4f}, Val Recall: {val_recall:.4f}')
        print(f'Train F1: {train_f1:.4f}, Val F1: {val_f1:.4f}')

        if val_loss < best_val_loss:
            best_val_loss = val_loss
            no_improve_count = 0  # Reset counter when validation loss improves
            torch.save(model.state_dict(), r'C:\Users\ssy\Desktop\111\BERT_CNN_BiLSTM_best3.pth')
            print("Best model saved.")
        else:
            no_improve_count += 1  # Increase counter when validation loss does not improve
            if no_improve_count == 3:  # Stop training when validation loss does not improve for 3 epochs
                print("Early stopping triggered.")
                break

    print("Training finished.")


if __name__ == '__main__':
    main()
